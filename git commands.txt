place=$1
date=$2
user=$(whoami)		#whoami is a command the output of this command will be stored in user variable

echo "hello $name"	
sleep 1			#sleep for 1 second

echo "you entered $place"
sleep 1

echo "you entered $date"
sleep 1

echo "you are $user"
